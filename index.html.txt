<script>
function search(){
    let input=document.querySelector("input").value.toLowerCase();
    let rows=document.querySelectorAll("#table tr");

    rows.forEach((row,i)=>{
        if(i===0) return;
        let text=row.innerText.toLowerCase();
        row.style.display=text.includes(input)?"":"none";
    });
}
